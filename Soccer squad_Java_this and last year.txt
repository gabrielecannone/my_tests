package verifica28032022;

public class Calciatore 
{
	private String nome, cognome;
	private int goal;
	
	public Calciatore(String n, String c, int g) 
	{
		nome = n;
		cognome = c;
		goal = g;
	}
	
	public String getNome() 
	{
		return nome;
	}
	public String getCognome() 
	{
		return cognome;
	}
	public int getGoal() 
	{
		return goal;
	}

}



















package verifica28032022;

public class Portiere extends Calciatore
{
	private int goal_in;
	
	public Portiere(String n, String c, int g, int i) 
	{
		super(n, c, g);
		goal_in = i;
	}
	
	public int getGoalSubiti() 
	{
		return goal_in;
	}

}
































package verifica28032022;

public class Squadra 
{
	private Calciatore calciatori[];
	private Portiere portieri[];
	public Squadra(int c, int p) 
	{
		calciatori = new Calciatore[c];
		portieri = new Portiere[p];
	}
	
	public void AggiungCalciatore(int i, String n, String c, int g) 
	{
		if (g>-1) 
		{
			calciatori[i]= new Calciatore(n, c, g);
		}
		
	}
	
	public void AggiungiPortiere(int i, String n, String c, int g, int g_in) 
	{
		if (g>-1 && g_in>-1) 
		{
			portieri[i]= new Portiere(n, c, g, g_in);
		}
		
	}
	
	public int goalFatti() 
	{
		int i;
		int med = 0;
		for(i=0; i<calciatori.length; i++) 
		{
			med += calciatori[i].getGoal();
		}
		for(i=0; i<portieri.length; i++) 
		{
			med += portieri[i].getGoal();
		}

		return med;
	}
	
	public int goalSubiti() 
	{
		int i;
		int med = 0;
		for(i=0; i<portieri.length; i++) 
		{
			med += portieri[i].getGoalSubiti();
		}
		return med;
	}
	
	public void StampaSquadra() 
	{
		int i;
		for (i=0; i<calciatori.length; i++) 
		{
			System.out.println("Calciatore "+calciatori[i].getNome()+" "+calciatori[i].getCognome()+" con goal: "+calciatori[i].getGoal());
		}
		for (i=0; i<portieri.length; i++) 
		{
			System.out.println("Portiere "+portieri[i].getNome()+" "+portieri[i].getCognome()+" con goal: "+portieri[i].getGoal()+" e goal subiti: "+portieri[i].getGoalSubiti());
		}
	}

}
